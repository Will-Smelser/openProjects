127 	/**
128 	 * If previous, next, first, last buttons are being used this will be called to update the pagination on each page click.
129 	 * This has not been tested in large page ques.
130 	 * @return void
131 	 */
132 	setPaginationDisplay : function(){
133 		var objRef = this;
134 
135 		//current pages index
136 		var index  = (typeof this.info.cache.currentIndex == 'undefined') ? 0 : this.info.cache.currentIndex;
137 		
138 		//max number to display
139 		var max = Math.min(this.info.defaults.maxShow,this.pageInfo.length);
140 		
141 		//figure out how many pages show before and after
142 		if(index == 0){
143 			var after = max - 1;
144 			var before = 0;
145 		}else if(index == this.pageInfo.length-1){
146 			var after = 0;
147 			var before = max - 1;
148 		}else{
149 			var goal = Math.floor(this.info.defaults.maxShow/2);
150 			var after = Math.min(goal,this.pageInfo.length-index-1);
151 			var before = max-after-1;
152 		}
153 		
154 		var start  = Math.max(index-before,0);
155 		var end    = Math.min(this.pageInfo.length-1,index+after);
156 		
157 		(start == 0) ? $('#'+this.info.defaults.lessId).addClass('hidden') : $('#'+this.info.defaults.lessId).removeClass('hidden');
158 		(end   == this.pageInfo.length-1) ? $('#'+this.info.defaults.moreId).addClass('hidden') : $('#'+this.info.defaults.moreId).removeClass('hidden');
159 		
160 		
161 		var $LIs = $('#'+this.manipulate.idHeader+' LI');		
162 		var sliceStart = (this.info.defaults.FirstLast) ? 2 : 1; //start at 1 since we have the '...' by default
163 		sliceStart += (this.info.defaults.PrevNext) ? 1 : 0;
164 		
165 		var i=0;
166 		$LIs.slice(sliceStart,-sliceStart).each(function(index,el){
167 			//reset
168 			$(el).removeClass('hidden');
169 			
170 			//hide seperator
171 			if($(el).hasClass('seperate') && i<=start && i != 0){
172 				$(el).addClass('hidden');
173 			}
174 			//hide/show pages
175 			var $a = $(el).children().first(); 
176 			if(!$a.hasClass('mioSkip') && $a.hasClass('page')){
177 				var page = objRef.pageInfo[i];
178 				(i < start || i > end) ? $(el).addClass('hidden') : $(el).removeClass('hidden');
179 				i++;
180 			}
181 			//hide show trailing seperator
182 			if($(el).hasClass('seperate') && i > end){
183 				$(el).addClass('hidden');
184 			}
185 		});
186 	},

320 		 * @public

323 		 * @param {Funciton} callback [optional] Function to be called after the open event occurrs
324 		 */
325 		mainHide : function($target,callback){
326 			$target = $target; 
327 
328 			//bail if simple animation
329 			if(this.objRef.info.mode == 'simple'){ return;}
330 			
331 			//if center animation
332 			if(this.objRef.info.mode.indexOf('center') >= 0 && this.objRef.info.mode.indexOf('animate') >=0){
333 				var ht = (this.objRef.info.cache.paginateHeight)/2;

336 			}
337 			
338 			if(this.objRef.info.mode.indexOf('animate') >= 0){
339 				$target.slideUp(speed);
340 			}else{
341 				$target.hide();
342 			}
343 			
344 			if(typeof callback =='function'){
345 				setTimeout(callback,100);
346 			}
347 			
348 		},
349 		/**
350 		 * Show the main content box
351 		 * @param {Object} $target Jquery object reference of the iframe or div box DOM element
352 		 * @param {Function} callback [optional] Function to be called at the end
353 		 */
354 		mainShow : function($target, callback){
355 			var objRef = this.objRef;
356 			if(this.objRef.info.mode == 'simple'){ return;}
357 			
358 			var speed = this.animSpeed;
359 			var slid  = false;
360 			var $main  = $target;//$('#'+this.idMain);
361 			
362 			//if center animation
363 			if(this.objRef.info.mode.indexOf('center') >= 0 && this.objRef.info.mode.indexOf('animate') >=0){
364 				var ht = (this.objRef.info.cache.footHeight+this.objRef.info.cache.paginateHeight+this.objRef.info.cache.mainHeight)/2;
365 				if(this.objRef.info.mode.indexOf('bounce') >= 0){
366 					slid = true;
367 					$('#'+this.idWrapper).animate({marginTop : -ht},speed,function(){$main.slideDown(speed);});
368 				}else{
369 					$('#'+this.idWrapper).animate({marginTop : -ht},speed);
370 				}
371 			}
372 			
373 			//show box
374 			if(this.objRef.info.mode.indexOf('animate') >= 0){
375 				if(!slid){ $main.slideDown(speed);}
376 			}else if(this.objRef.info.mode.indexOf('animate') < 0){
377 				$main.show();
378 			}
379 			
380 			//use the fade-in animation
381 			if(this.objRef.info.mode.indexOf('fade') >= 0){
382 				$main.css({opacity:0});
383 				$main.animate({opacity:objRef.info.opacity},speed*2);
384 			}
385 
386 			if(typeof callback =='function'){
387 				setTimeout(callback,100);
388 			}

579 		_fireClick : function(page){
580 			//if we are at the current page, dont run events
581 			if(page.name == this.objRef.info.cache.currentPage){return;}
582 			
583 			//handle first click
584 			var event  = (!page['clicked']) ? 'firstClick' : 'notFirstClick';
585 			page.eventParams.mioEach(event);
586 			
587 			//first click and backwards navigation
588 			if(page['clicked']){
589 				page.eventParams.mioEach('navBackwards');
590 				if(!page['allowBackNav']){return;}
591 			}			
592 			
593 			//sequential events
594 			for(var x in this.objRef.pageInfo){
595 				if(this.objRef.pageInfo[x].name == page.name) break; //found the key of the current page
596 			}
597 			var bad = false;
598 			if(x > 0){
599 				//find this pages position
600 				for(var i=x-1; i>=0; i--){
601 					if(this.objRef.pageInfo[i].clicked === false){bad = true;break;}
602 				}
603 			}
604 			
605 			//if this must be performed in sequence
606 			if(bad){
607 				page.eventParams.mioEach('notSequential');
608 				if(page['sequential']){ return;}
609 			}	
610 			
611 			//standard events
612 			page['clicked'] = true;
613 			page.eventParams.mioEach('beforeClick');
614 			page.eventParams.mioEach('onClick');
615 			page.eventParams.mioEach('afterClick');
616 			
617 			this.objRef.setPaginationDisplay();
618 			
619 			return;
620 			
621 		},

874 	_orderObject : function(origArr,newArr,fieldName){
875 		if(typeof origArr == 'undefined') return [];
876 
877 		var tmp = origArr.pop();
878 		
879 		//just make sure we did enter something
880 		var entered =  false;
881 		
882 		//cycle the new Array looking for where to insert the new value
883 		for(var x in newArr){
884 			var prev = (x > 0) ? newArr[x-1][fieldName] : 0;
885 			
886 			//the value is between last value and current value....so enter befor this value
887 			if(tmp[fieldName] <= newArr[x][fieldName] && tmp[fieldName] >= prev){
888 				//now we need to splice this into the array
889 				newArr.splice(x,0,tmp);
890 				entered = true;
891 				break;
892 			}
893 		}
894 		
895 		//make sure we dont loose any values
896 		if(!entered){
897 			this.err.log('failed to enter an item.  Force entering.');
898 			newArr.push(tmp);
899 		}
900 		
901 		if(origArr.length > 0){
902 			this._orderObject(origArr,newArr,fieldName);
903 		}
904 
905 		return newArr;
906 	},
