HashRouter Output

Sample Code:

var router = new HashRouter();

//add a route
router.addPath("/user/{name:[\\w]+}/{action}",function(data){
    window.myform.elements.output.value = JSON.stringify(data);
});

//add a route
router.addPath("/example/{id:[\\d]+}/and/{name}",function(data){
    window.myform.elements.output.value = JSON.stringify(data);
});

//override the noRoute handler
router.noRoute = function(path){
    window.myform.elements.output.value = "No Path Matched - " + path;
};

//tell the router to listen for "onhashchange" events
router.start();
